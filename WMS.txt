function useDebounce(value, delay) {
    const [debouncedValue, setDebouncedValue] = useState(value);
    useEffect(() => { const handler = setTimeout(() => { setDebouncedValue(value); }, delay); return () => { clearTimeout(handler); }; }, [value, delay]);
    return debouncedValue;
}

    const playSound = (type) => {
        if (!ttsEnabled) return;
        try {
            if (!audioCtx.current) audioCtx.current = new (window.AudioContext || window.webkitAudioContext)();
            const osc = audioCtx.current.createOscillator();
            const gain = audioCtx.current.createGain();
            osc.connect(gain); gain.connect(audioCtx.current.destination);
            if (type === 'success') {
                osc.type = 'sine';
                osc.frequency.setValueAtTime(880, audioCtx.current.currentTime);
                osc.frequency.exponentialRampToValueAtTime(440, audioCtx.current.currentTime + 0.1);
                gain.gain.setValueAtTime(0.1, audioCtx.current.currentTime);
                gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.current.currentTime + 0.1);
                osc.start(); osc.stop(audioCtx.current.currentTime + 0.1);
            } else {
                osc.type = 'sawtooth';
                osc.frequency.setValueAtTime(150, audioCtx.current.currentTime);
                osc.frequency.linearRampToValueAtTime(100, audioCtx.current.currentTime + 0.3);
                gain.gain.setValueAtTime(0.2, audioCtx.current.currentTime);
                gain.gain.linearRampToValueAtTime(0.01, audioCtx.current.currentTime + 0.3);
                osc.start(); osc.stop(audioCtx.current.currentTime + 0.3);
                if (navigator.vibrate) navigator.vibrate([100, 50, 100]);
            }
        } catch (e) { console.error("Audio failed", e); }

        }

        if (foundIndex !== -1) {
            const found = orderData[foundIndex];
            const newData = [...orderData];
            if (newData[foundIndex].scannedQty < newData[foundIndex].expectedQty) {
                newData[foundIndex].scannedQty += 1;
                newData[foundIndex].lastScannedAt = new Date().toLocaleTimeString();
                
                updateAndSyncData(newData);
                
                logUserAction('scan', found.orderNumber || code, code);
                
                setScanStatus({ status: 'success', message: `Scan Verified` });
                playSound('success');

                                    {selectedPackOrder.status === 'packed' ? (

                                            <p className="text-sm font-medium text-slate-500">Box Barcode: <strong className="font-mono text-slate-800 dark:text-slate-200 bg-slate-100 dark:bg-zinc-800 px-2.5 py-1 rounded-md border border-slate-200 dark:border-zinc-700 ml-1">{selectedPackOrder.boxBarcode}</strong></p>

                                                <div className="text-[10px] font-bold text-slate-400 uppercase tracking-widest mb-3">Next Action Steps</div>

                                                    <li className="flex items-center gap-2"><div className="w-1.5 h-1.5 rounded-full bg-primary-500"></div> Trigger RTS on {selectedPackOrder.platform || 'Platform'}</li>

                                                <button

                                                    disabled={isProcessingAPI}

                                                >
                                                    {isProcessingAPI ? (
                                                        <><RefreshCw className="w-4 h-4 animate-spin"/> Processing API Workflow...</>
                                                    ) : (
                                                        <><Printer className="w-4 h-4"/> Confirm, RTS & Print AWB</>
                                                    )}
                                                </button>
                                            </div>
                                        </div>
                                    ) : (
                                        <div className="p-6 sm:p-8">
                                            {selectedPackOrder.items.every(i => i.packed === i.picked) ? (
                                                <div className="bg-slate-50/50 dark:bg-zinc-800/30 border-2 border-dashed border-slate-200 dark:border-zinc-700 p-10 rounded-2xl text-center animate-fade-in max-w-md mx-auto">
                                                    <Barcode className="w-12 h-12 text-slate-400 mx-auto mb-5"/>
                                                    <h3 className="text-base font-bold text-slate-800 dark:text-slate-200 mb-6">Items verified. Scan box barcode to close.</h3>
                                                    <div className="relative group">
                                                        <div className="absolute inset-0 bg-slate-500/10 rounded-xl blur-lg group-focus-within:bg-slate-500/20 transition-all opacity-0 group-focus-within:opacity-100"></div>
                                                        <input ref={boxInputRef} type="text" id="box-scan" value={boxBarcode} onChange={e=>setBoxBarcode(e.target.value)} onKeyDown={handleBoxBarcodeSubmit} placeholder="Scan Box..." className="relative w-full bg-white dark:bg-zinc-950 border-2 border-slate-200 dark:border-zinc-700 focus:border-slate-500 focus:ring-4 focus:ring-slate-500/10 rounded-xl text-lg font-mono font-bold text-center py-4 outline-none transition-all"/>
                                                    </div>
                                                </div>
                                            ) : (
                                                <div className="max-w-md mx-auto mb-8 relative group">
                                                    <div className="absolute inset-0 bg-purple-500/20 rounded-xl blur-lg group-focus-within:bg-purple-500/30 transition-all opacity-0 group-focus-within:opacity-100"></div>
                                                    <input ref={packInputRef} type="text" autoFocus id="pack-scan" value={packScanInput} onChange={e=>setPackScanInput(e.target.value)} onKeyDown={handlePackScanSubmit} placeholder="Scan SKU to verify..." className="relative w-full bg-white dark:bg-zinc-950 border-2 border-slate-200 dark:border-zinc-800 focus:border-purple-500 focus:ring-4 focus:ring-purple-500/10 rounded-xl text-lg font-mono font-bold text-center py-4 outline-none transition-all shadow-sm"/>
                                                </div>
                                            )}
                                            <div className="mt-8 space-y-3">
                                                {selectedPackOrder.items.map((item, idx) => {
                                                    const isComplete = item.packed === item.picked;
                                                    return (
                                                        <div key={idx} className={`p-4 rounded-xl border-2 flex items-center justify-between transition-all ${isComplete ? 'bg-purple-50/50 border-purple-200/60 dark:bg-purple-900/10 dark:border-purple-800/40' : 'bg-white dark:bg-zinc-900 border-slate-100 dark:border-zinc-800'}`}>
                                                            <div className="flex items-center gap-4">
                                                                <div className={`p-2.5 rounded-lg ${isComplete ? 'bg-purple-100 text-purple-600 dark:bg-purple-900/40 dark:text-purple-400' : 'bg-slate-100 text-slate-500 dark:bg-zinc-800 dark:text-slate-400'}`}>{isComplete ? <CheckSquare className="w-5 h-5"/> : <Box className="w-5 h-5"/>}</div>
                                                                <div><p className={`font-semibold text-sm ${isComplete ? 'text-purple-800 dark:text-purple-300' : 'text-slate-800 dark:text-slate-200'}`}>{item.name}</p><p className="text-[11px] font-mono font-medium text-slate-500 mt-1 uppercase tracking-wider">SKU: {item.sku}</p></div>
                                                            </div>
                                                            <div className="text-right"><span className={`text-2xl font-black tabular-nums ${isComplete ? 'text-purple-600 dark:text-purple-400' : 'text-slate-800 dark:text-white'}`}>{item.packed}</span><span className="text-sm font-bold text-slate-400 mx-1">/</span><span className="text-lg font-bold text-slate-400 tabular-nums">{item.picked}</span></div>
                                                        </div>
                                                    );
                                                })}
                                            </div>
                                        </div>
                                    )}
                                </div>
                            )}
                        </div>
                    )}

                    {/* SCAN TAB (OUTBOUND) */}
                    {activeTab === 'scan' && (
                        <div className="max-w-3xl mx-auto flex flex-col items-center animate-slide-up pt-4 pb-20">
                            <div className="w-full mb-6 flex items-center justify-between">
                                <div className="text-xs font-semibold text-slate-500 uppercase tracking-widest flex items-center gap-2">Batch <span className="text-slate-800 dark:text-slate-200 font-mono bg-white dark:bg-zinc-800 border border-slate-200 dark:border-zinc-700 px-2.5 py-1 rounded-md shadow-sm">{currentBatchId||'READY'}</span></div>
                                <div className="text-xs font-semibold text-slate-500 uppercase tracking-widest flex items-center gap-2">Progress <span className="text-primary-600 dark:text-primary-400 font-black text-base">{totalScanned}/{totalExpected}</span></div>
                            </div>
                            
                            <div className="w-full bg-white dark:bg-zinc-900 border border-slate-200/80 dark:border-zinc-800 rounded-3xl shadow-lg p-10 text-center relative overflow-hidden">
                                <div className="absolute top-0 left-0 w-full h-1.5 bg-slate-100 dark:bg-zinc-800"><div className="h-full bg-primary-500 transition-all duration-500 ease-out" style={{width:`${progressPercent}%`}}></div></div>
                                <h2 className="text-[11px] font-bold text-slate-400 dark:text-zinc-500 uppercase tracking-[0.2em] mb-8 mt-2 flex items-center justify-center gap-2"><ScanLine className="w-4 h-4"/> Scan AWB Barcode</h2>
                                
                                <div className="relative max-w-lg mx-auto">
                                    <input ref={inputRef} type="text" autoFocus id="main-scan-input" value={scanInput} onChange={e=>setScanInput(e.target.value)} onKeyDown={handleScan} placeholder="SCAN..." className="industrial-input w-full"/>
                                    <div className="absolute inset-x-0 bottom-0 h-[4px] bg-primary-500/20 blur-[6px] pointer-events-none"></div>
                                </div>
                                
                                {scanBinHint && (
                                    <div className={`mt-10 p-8 rounded-2xl ${scanBinHint.color} ${scanBinHint.border} border-2 text-white animate-slide-up shadow-xl relative overflow-hidden group`}>
                                        <div className="absolute inset-0 bg-white/10 opacity-0 group-hover:opacity-100 transition-opacity"></div>
                                        <div className="relative z-10">
                                            <div className="text-xs font-black uppercase tracking-[0.2em] opacity-90 mb-3 flex items-center justify-center gap-2"><Truck className="w-4 h-4"/> {scanBinHint.label}</div>
                                            <div className="text-7xl font-black tracking-tighter drop-shadow-md">BIN {scanBinHint.bin}</div>
                                        </div>
                                    </div>
                                )}
                            </div>

                            <div className="w-full mt-10">
                                <h3 className="text-[10px] font-bold text-slate-400 dark:text-zinc-500 uppercase tracking-[0.2em] mb-4 px-2">Recent Scans</h3>
                                <div className="space-y-2.5">
                                    {lastScans.length>0 ? lastScans.map((item,i) => (
                                        <div key={i} className={`flex items-center justify-between p-4 rounded-xl border bg-white dark:bg-zinc-900 shadow-sleek transition-all ${item.status==='OK'?'border-l-4 border-l-emerald-500 border-slate-200 dark:border-zinc-800':'border-l-4 border-l-red-500 border-red-200 dark:border-red-900/50'}`}>
                                            <div className="flex items-center gap-4"><div className={item.status==='OK'?'text-emerald-500':'text-red-500'}>{item.status==='OK'?<CheckCircle2 className="w-5 h-5"/>:<XCircle className="w-5 h-5"/>}</div><span className="font-mono font-bold text-base text-slate-800 dark:text-slate-100 tracking-tight">{item.barcode}</span></div>
                                            <span className="text-xs font-semibold text-slate-400 bg-slate-50 dark:bg-zinc-800 px-2 py-1 rounded">{item.time}</span>

                                    )) : <div className="text-center py-12 text-slate-400 text-sm font-medium border-2 border-dashed border-slate-200 dark:border-zinc-800 rounded-xl bg-white/50 dark:bg-zinc-900/50">Waiting for scan...</div>}

                        </div>
                    )}

                    {/* LIST TAB */}
                    {activeTab === 'list' && (
                        <div className="h-[calc(100vh-8rem)] flex flex-col bg-white dark:bg-zinc-900 border border-slate-200/80 dark:border-zinc-800 rounded-2xl shadow-sleek overflow-hidden animate-slide-up">
                            <div className="p-4 border-b border-slate-200/80 dark:border-zinc-800 bg-slate-50/50 dark:bg-zinc-900/50 flex flex-wrap gap-4 justify-between items-center shrink-0">

                                <table className="w-full text-left text-sm border-collapse">
                                    <thead className="bg-slate-50/90 dark:bg-zinc-900/90 backdrop-blur-md text-xs font-bold text-slate-500 dark:text-slate-400 uppercase tracking-wider sticky top-0 z-10 shadow-sm">

                                        {paginatedListData.map((item, idx) => (

                                                    {item.scannedQty >= item.expectedQty ? <span className="inline-flex items-center gap-1.5 text-emerald-600 dark:text-emerald-400 font-bold text-xs bg-emerald-50 dark:bg-emerald-900/20 px-2.5 py-1 rounded-md"><CheckCircle2 className="w-3.5 h-3.5" /> Scanned</span> : <span className="inline-flex items-center gap-1.5 text-slate-500 font-semibold text-xs bg-slate-100 dark:bg-zinc-800 px-2.5 py-1 rounded-md"><Clock className="w-3.5 h-3.5" /> Pending</span>}

                                            </tr>
                                        ))}
                                        {paginatedListData.length === 0 && <tr><td colSpan="4" className="text-center py-16 text-slate-400 font-medium"><div className="flex flex-col items-center"><Search className="w-8 h-8 mb-3 opacity-20"/> No matching records found</div></td></tr>}

                                </table>
                            </div>
                            {totalPages > 1 && (
                                <div className="p-4 border-t border-slate-200/80 dark:border-zinc-800 bg-white dark:bg-zinc-900 flex justify-between items-center shrink-0">
                                    <div className="text-xs font-medium text-slate-500">Showing {((currentPage - 1) * ITEMS_PER_PAGE) + 1}-{Math.min(currentPage * ITEMS_PER_PAGE, filteredListData.length)} of {filteredListData.length}</div>

                                        <span className="px-3 py-1.5 text-xs font-bold bg-slate-50 dark:bg-zinc-800 rounded-md border border-slate-200 dark:border-zinc-700">{currentPage} / {totalPages}</span>
                                        <button onClick={() => setCurrentPage(p => Math.min(totalPages, p + 1))} disabled={currentPage === totalPages} className="p-1.5 rounded-lg text-slate-500 hover:bg-slate-100 dark:hover:bg-zinc-800 disabled:opacity-30 transition-colors"><ChevronRight className="w-5 h-5" /></button>
                                    </div>
                                </div>
                            )}
                        </div>
                    )}

                    {/* DISPATCH TAB */}
                    {activeTab === 'dispatch' && (
                        <div className="max-w-3xl mx-auto animate-slide-up pb-12">
                            <div className="bg-white dark:bg-zinc-900 border border-slate-200/80 dark:border-zinc-800 rounded-2xl shadow-sleek overflow-hidden">
                                <div className="px-6 py-5 border-b border-slate-200/80 dark:border-zinc-800 bg-slate-50/50 dark:bg-zinc-900/50">
                                    <h2 className="text-lg font-bold text-slate-900 dark:text-white flex items-center gap-2 tracking-tight"><Truck className="w-5 h-5 text-primary-500" /> Carrier Dispatch</h2>
                                    <p className="text-xs font-medium text-slate-500 mt-1">Hand over scanned packages to courier drivers.</p>
                                </div>
                                <div className="p-6 sm:p-8">
                                    {readyToDispatchCouriers.length === 0 ? (
                                        <div className="text-center py-16 text-slate-400 flex flex-col items-center"><Package className="w-12 h-12 mb-4 opacity-20" /><p className="text-sm font-medium">No packages ready for dispatch</p></div>
                                    ) : (
                                        <div className="space-y-8">
                                            <div>
                                                <label className="text-xs font-bold text-slate-400 uppercase tracking-widest mb-3 block">1. Select Courier</label>
                                                <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
                                                    {readyToDispatchCouriers.map((c) => (
                                                        <button key={c.name} onClick={() => setDispatchCourier(c.name)} className={`p-4 rounded-xl font-bold text-sm transition-all border-2 flex flex-col items-center gap-2.5 ${dispatchCourier === c.name ? 'border-primary-500 bg-primary-50 text-primary-700 dark:bg-primary-900/20 dark:text-primary-400 shadow-sm' : 'border-slate-100 dark:border-zinc-800 bg-white dark:bg-zinc-900 text-slate-600 dark:text-slate-400 hover:border-slate-300 dark:hover:border-zinc-600'}`}>
                                                            <span className="truncate w-full text-center tracking-tight">{c.name}</span>
                                                            <span className={`text-[10px] uppercase tracking-wider px-2 py-0.5 rounded-md ${dispatchCourier === c.name ? 'bg-primary-100 dark:bg-primary-900/50' : 'bg-slate-100 dark:bg-zinc-800'}`}>{c.count} pkgs</span>
                                                        </button>
                                                    ))}
                                                </div>
                                            </div>
                                            <div className="space-y-3">
                                                <div className="flex justify-between items-end">
                                                    <label className="text-xs font-bold text-slate-400 uppercase tracking-widest block">2. Driver Signature</label>
                                                    <button onClick={clearSignature} className="text-xs font-bold text-slate-500 hover:text-slate-900 dark:hover:text-white flex items-center gap-1 transition-colors bg-slate-100 dark:bg-zinc-800 px-2 py-1 rounded"><Eraser className="w-3.5 h-3.5" /> Clear</button>
                                                </div>
                                                <div className="relative group">
                                                    <canvas ref={canvasRef} onMouseDown={startDrawing} onMouseMove={draw} onMouseUp={endDrawing} onMouseOut={endDrawing} onTouchStart={startDrawing} onTouchMove={draw} onTouchEnd={endDrawing} className="signature-canvas" />
                                                    {signatureEmpty && (
                                                        <div className="absolute inset-0 pointer-events-none flex flex-col items-center justify-center text-slate-400 opacity-50">
                                                            <PenLine className="w-8 h-8 mb-3" />
                                                            <span className="text-sm font-semibold">Driver: Sign here</span>
                                                        </div>
                                                    )}
                                                </div>
                                            </div>
                                            <div className="pt-6 border-t border-slate-200/80 dark:border-zinc-800">
                                                <button onClick={handleDispatchSubmit} disabled={!dispatchCourier || signatureEmpty} className="w-full py-4 rounded-xl bg-slate-900 hover:bg-slate-800 dark:bg-white dark:hover:bg-slate-200 text-white dark:text-slate-900 font-bold shadow-lg transition-all flex items-center justify-center gap-2 disabled:opacity-50 disabled:cursor-not-allowed">
                                                    Confirm Dispatch <span className="opacity-70 font-normal">({readyToDispatchCouriers.find(c => c.name === dispatchCourier)?.count || 0} items)</span>
                                                </button>
                                            </div>
                                        </div>
                                    )}
                                </div>
                            </div>
                        </div>
                    )}

                    {/* Report Tab */}
                    {activeTab === 'report' && (() => {
                      let displayReportData = {};
                      let currentReportOrderId = orderId;

                      if (reportViewMode === 'current') {
                        const activeScans = orderData.filter(i => i.scannedQty > 0);

                        if (reportFilterCourier === 'All') {
                          displayReportData = activeScans.reduce((acc, curr) => {
                            const name = curr.courier || 'KISS';
                            if (!acc[name]) acc[name] = [];
                            acc[name].push(curr);
                            return acc;
                          }, {});
                        } else {
                          const filtered = activeScans.filter(
                            i => (i.courier || 'KISS') === reportFilterCourier
                          );

                          if (filtered.length > 0)
                            displayReportData = { [reportFilterCourier]: filtered };

                          currentReportOrderId =
                            courierBatches[reportFilterCourier] || orderId;
                        }
                      } else {
                        const selectedBatch = historyData.find(
                          h => h.id === reportFilterBatchId
                        );

                        if (selectedBatch && selectedBatch.items) {
                          currentReportOrderId = selectedBatch.id;

                          const batchItems = selectedBatch.items;

                          displayReportData = batchItems.reduce((acc, curr) => {
                            const name = curr.courier || 'KISS';
                            if (!acc[name]) acc[name] = [];
                            acc[name].push(curr);
                            return acc;
                          }, {});
                        }
                      }

                      return (
                        <div className="flex flex-col items-center animate-slide-up pb-10">
                          <div className="w-full max-w-[210mm] bg-white dark:bg-zinc-900 p-5 rounded-2xl shadow-sleek border border-slate-200/80 dark:border-zinc-800 mb-6 print-hide flex flex-wrap gap-5 items-end">

                <div className="fixed inset-0 bg-red-600 flex flex-col items-center justify-center p-6 z-[120] text-white text-center animate-pulse-fast">
                    <AlertTriangle className="w-24 h-24 mb-6 opacity-90 drop-shadow-md" />
                    <h1 className="text-5xl font-black mb-2 uppercase tracking-tight drop-shadow-md">Alert</h1>
                    <p className="text-2xl font-semibold mb-12 opacity-90">{scanStatus?.message}</p>
                    <button onClick={() => { setIsErrorLocked(false); setScanStatus(null); setScanBinHint(null); }} className="bg-white text-red-600 px-10 py-4 rounded-2xl text-lg font-black shadow-xl hover:scale-105 active:scale-95 transition-all flex items-center gap-3">
                        <Unlock className="w-6 h-6"/> Press Spacebar to Unlock
                    </button>
                </div>
            )}

            {confirmModal.open && (
                <div className="fixed inset-0 bg-slate-900/60 backdrop-blur-sm flex items-center justify-center p-4 z-[110] animate-fade-in">
                    <div className="bg-white dark:bg-zinc-900 rounded-2xl p-6 w-full max-w-sm shadow-2xl border border-slate-200/80 dark:border-zinc-800">
                        <div className="w-12 h-12 rounded-full flex items-center justify-center mb-4 bg-slate-100 dark:bg-zinc-800">
                            {confirmModal.type === 'danger' ? <AlertTriangle className="w-6 h-6 text-red-500"/> : <Info className="w-6 h-6 text-primary-500"/>}
                        </div>
                        <h3 className="text-lg font-bold tracking-tight text-slate-900 dark:text-white mb-2">{confirmModal.title}</h3>
                        <p className="text-slate-500 dark:text-slate-400 text-sm font-medium mb-8 leading-relaxed">{confirmModal.message}</p>
                        <div className="flex justify-end gap-3">
                            {!confirmModal.isAlert && <button onClick={() => setConfirmModal({ open: false })} className="px-5 py-2.5 rounded-xl text-sm font-bold text-slate-600 dark:text-slate-300 hover:bg-slate-100 dark:hover:bg-zinc-800 transition-colors">Cancel</button>}
                            <button onClick={() => { if (confirmModal.onConfirm) confirmModal.onConfirm(); setConfirmModal({ open: false }); }} className={`px-5 py-2.5 rounded-xl text-white text-sm font-bold shadow-md transition-colors ${confirmModal.type === 'danger' ? 'bg-red-600 hover:bg-red-700' : 'bg-slate-900 hover:bg-slate-800 dark:bg-white dark:text-slate-900'}`}>{confirmModal.isAlert ? 'Got it' : 'Confirm Action'}</button>
                        </div>
                    </div>
                </div>
            )}

            <div className="fixed bottom-6 right-6 z-[100] flex flex-col gap-3 pointer-events-none">
                {toasts.map(t => (
                    <div key={t.id} className={`animate-slide-in-right flex items-center gap-3 px-5 py-3.5 rounded-xl shadow-xl text-sm font-bold border ${t.type === 'error' ? 'bg-white dark:bg-zinc-900 text-red-600 dark:text-red-400 border-red-200 dark:border-red-900/50' : 'bg-slate-900 dark:bg-white text-white dark:text-slate-900 border-transparent'}`}>
                        {t.type === 'error' ? <AlertCircle className="w-5 h-5"/> : <CheckCircle2 className="w-5 h-5 text-emerald-400 dark:text-emerald-500"/>}
                        {t.msg}
                    </div>
                ))}
            </div>
        </div>
    );
};

createRoot(document.getElementById('root')).render(<App />);
</script>
